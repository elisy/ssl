Процедура ОбработкаКоманды(ПараметрКоманды, ПараметрыВыполненияКоманды)
	
	Если ПараметрКоманды = Неопределено
		ИЛИ ПараметрКоманды.Пустая() Тогда
		Возврат;
	КонецЕсли;
	
	ПараметрыФормы = Новый Структура;
	IDПредметаЗадач = Новый Массив;
	IDПредметаЗадач.Добавить(Строка(ПараметрКоманды.УникальныйИдентификатор()));
	ПараметрыФормы.Вставить("IDПредметаЗадач", IDПредметаЗадач);
	ПараметрыФормы.Вставить("НаименованиеПредметаЗадач", Строка(ПараметрКоманды));
	ПараметрыФормы.Вставить("СсылкаНаВнешнийОбъект", ПараметрКоманды);
	ОткрытьФорму("ОбщаяФорма.ДокументооборотЗадачи", ПараметрыФормы, ПараметрыВыполненияКоманды.Источник, ПараметрыВыполненияКоманды.Уникальность, ПараметрыВыполненияКоманды.Окно);
	 
КонецПроцедуры
