&НаСервере
Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	
	Если Параметры.Свойство("АвтоТест") Тогда
		Возврат;
	КонецЕсли;
	
	
	СоединенияИнфоБазы = ПолучитьСоединенияИнформационнойБазы();
	
	Если СоединенияИнфоБазы = Неопределено Тогда
		Элементы.КнопкаАктивныхПользователей.Заголовок = 1; 
	Иначе
		Элементы.КнопкаАктивныхПользователей.Заголовок = СоединенияИнфоБазы.Количество();
	КонецЕсли;
	
	СписокПроверок = Параметры.СписокПроверок;
	
		
КонецПроцедуры
