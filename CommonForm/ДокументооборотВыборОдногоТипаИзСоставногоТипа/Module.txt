
&НаСервере
Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	
	Если ЗначениеЗаполнено(Параметры.ЗаголовокФормы) Тогда
		АвтоЗаголовок = Ложь;
		Заголовок = Параметры.ЗаголовокФормы;
	КонецЕсли;
	
	Если Параметры.СписокДоступныхТипов.Количество() = 0 Тогда
		Отказ = Истина;
		Возврат;
	КонецЕсли;
	
	Для Каждого Элемент Из Параметры.СписокДоступныхТипов Цикл
		НоваяСтрока = ДоступныеТипы.Добавить();
		НоваяСтрока.Представление = Элемент.Значение.Presentation;
		НоваяСтрока.Имя = Элемент.Значение.XDTOClassName;
	КонецЦикла;
	
КонецПроцедуры
