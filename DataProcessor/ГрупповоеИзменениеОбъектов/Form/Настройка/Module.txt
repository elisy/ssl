&НаКлиенте
Процедура УстановитьЭлементыФормы()
	
	Если ИзменятьВТранзакции Тогда
		Элементы.ГруппаПрерываниеПриОшибке.Доступность = Ложь;
	Иначе
		Элементы.ГруппаПрерываниеПриОшибке.Доступность = Истина;
	КонецЕсли;
	
КонецПроцедуры

// Блок процедур - обработчиков команд формы

&НаКлиенте
Процедура ОК(Команда)
	
	Настройки = Новый Структура;
	Настройки.Вставить("ИзменятьВТранзакции",		ИзменятьВТранзакции);
	Настройки.Вставить("ОбрабатыватьРекурсивно",	ОбрабатыватьРекурсивно);
	Настройки.Вставить("НастройкаПорции",			НастройкаПорции);
	Настройки.Вставить("ПроцентОбъектовВПорции",	ПроцентОбъектовВПорции);
	Настройки.Вставить("ЧислоОбъектовВПорции",		ЧислоОбъектовВПорции);
	Настройки.Вставить("ПрерыватьПриОшибке",		ПрерыватьПриОшибке);
	
	Закрыть(Настройки);
	
КонецПроцедуры

Процедура ПриСозданииНаСервере(Отказ, СтандартнаяОбработка)
	
	ИзменятьВТранзакции			= Параметры.ИзменятьВТранзакции;
	ОбрабатыватьРекурсивно		= Параметры.ОбрабатыватьРекурсивно;
	НастройкаПорции				= Параметры.НастройкаПорции;
	ПроцентОбъектовВПорции		= Параметры.ПроцентОбъектовВПорции;
	ЧислоОбъектовВПорции		= Параметры.ЧислоОбъектовВПорции;
	ПрерыватьПриОшибке			= Параметры.ПрерыватьПриОшибке;
	
КонецПроцедуры
